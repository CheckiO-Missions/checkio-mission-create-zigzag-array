def create_zigzag(rows: int, cols: int, start: int = 1) -> list[list[int]]:

    grid = []
    for row_ind in range(rows):
        row = []
        for _ in range(cols):
            row.append(start)
            start += 1
        if row_ind % 2:
            row.reverse()
        grid.append(row)

    return grid
